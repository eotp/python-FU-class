## Plot
def attk_seasons(results, category_names):
    labels = list(results.keys()) # List the (Season-)labels
    data = np.array(list(results.values())) # List the values of seasons
    data_cum = data.cumsum(axis=1) # Sum of Data on axis
    category_colors = plt.get_cmap('RdYlGn')(np.linspace(0.15, 0.85, data.shape[1]))

    fig, ax = plt.subplots()
    ax.invert_yaxis()
    ax.xaxis.set_visible(False) # No visable x-Axis
    ax.set_xlim(0, np.sum(data, axis=1).max())

    for i, (colname, color) in enumerate(zip(category_names, category_colors)):
        widths = data[:, i]
        starts = data_cum[:, i] - widths
        ax.barh(labels, 
                widths, 
                left=starts, 
                height=0.5,
                label=colname, 
                color=color)
        xcenters = starts + widths / 2

        r, g, b, _ = color # Split color values
        text_color = 'white' if r * g * b < 0.5 else 'darkgrey' # If text-color is too light, then grey
        
        for y, (x, c) in enumerate(zip(xcenters, widths)):
            if c < 1:
                ax.text(x, y, str(int(c)), 
                        ha='center', 
                        va='center',
                        color=text_color).set_alpha(0) # Make Zeros invisible
            else:
                ax.text(x, y, str(int(c)), 
                        ha='center', 
                        va='center',
                        color=text_color).set_alpha(0.9)
    ax.legend(ncol=len(category_names), 
              bbox_to_anchor=(0, 1),
              loc='lower left', 
              fontsize='small') # Legend customization

    return fig, ax
